    $(function () {
        var startWithMap = "offices";
        var map;

        function initMap() {
            map = new L.Map('osmMap');

            // create the tile layer with correct attribution
            var osmUrl = '//{s}.tile.openstreetmap.org/{z}/{x}/{y}.png';
            var osmAttrib = 'Map data  <a href="//openstreetmap.org">OpenStreetMap</a> contributors';
            var osm = new L.TileLayer(osmUrl, {minZoom: 0, maxZoom: 19, attribution: osmAttrib});

            map.setView(new L.LatLng(40.489, -3.693), 1);

            map.addLayer(osm);
        }

        function initOffices(map, officeLayer) {
            $.ajax({
                type: "GET",
                dataType: "json",
                cache: true,
                url: "/.rest/map/offices/v1/full",
                success: function (rawData) {
                    var officeMarkers = [];

                    for (var i in rawData) {

                        var data = rawData[i];

                        var cs = '<div id="infoWindow">';
                        //var cs = '';

                        if (data.name)
                            cs += '<b>' + data.name + '</b><br/>';
                        if (data.address1 && data.address1.length > 2)
                            cs += data.address1 + '<br/>';
                        if (data.address2 && data.address2.length > 2)
                            cs += data.address2 + '<br/>';

                        if (data.country === "Brazil") {
                            if (data.city && data.city.length > 2)
                                cs += data.city + "<br/>";
                            if (data.zipcode && data.zipcode.length > 2)
                                cs += data.zipcode;
                        }
                        else if (data.country === "United States"
                                || data.country === "Australia"
                                || data.country === "Canada"
                                || data.country === "Thailand"
                                || data.country === "Korea"
                                || data.country === "Vietnam"
                                || data.country === "Bangladesh"
                                || data.country === "India") {
                            if (data.city && data.city.length > 2)
                                cs += data.city;
                            if (data.state && data.state.length > 0)
                                cs += ', ' + data.state + ' ';
                            if (data.zipcode && data.zipcode.length > 2)
                                cs += data.zipcode;
                        } else {
                            if (data.zipcode && data.zipcode.length > 2)
                                cs += data.zipcode + ' ';
                            if (data.city && data.city.length > 2)
                                cs += data.city;
                        }

                        cs += '<br/><br/>';

                        if (data.phone && data.phone.length > 2)
                            cs += 'Phone: ' + data.phone + '<br/>';
                        if (data.fax && data.fax.length > 2)
                            cs += 'Fax: ' + data.fax + '<br/>';
                        if (data.email && data.email.length > 2)
                            cs += 'E-Mail: <a style="color: #009fda" href="mailto:' + data.email + '">' + data.email + '</a><br/>';
                        if (data.website && data.website.length > 2)
                            cs += 'Website: <a style="color: #009fda" href="' + data.website + '" target="_blank" rel="noopener">' + data.website + '</a><br/>';

                        /*
                        if (data.image && data.image.length > 2) {
                            theDescription = data.description;
                            theLink = data.link;
                            theImage = data.image;

                            cs += '<br/><a href="' + theLink + '"><img src="' + theImage + '"></a><br/>';
                        }*/

                        if (data.description && data.description.length > 2)
                            cs += '<i>' + theDescription + '</i><br/>';

                        cs += '</div>';

                        var officeMarker = L.marker([data.latitude, data.longitude]).bindPopup(cs);
                        officeMarkers.push(officeMarker);
                    }

                    // console.log("markers array: ", officeMarkers);
                    officeLayer = L.layerGroup(officeMarkers);
                    // console.log("layer: ", officeLayer);
                    map.addLayer(officeLayer);

                }
            });

        }

        function initCenters(map, officeLayer) {
            $.ajax({
                type: "GET",
                dataType: "json",
                cache: true,
                url: "/.rest/map/treatmentcenters/v1/full",
                success: function (rawData) {
                    var centerMarkers = [];

                    for (var i in rawData) {
                        var data = rawData[i];
                        var cs = '<div id="infoWindow">';

                        if (data.name)
                            cs += '<b>' + data.name + '</b><br/>';
                        if (data.address1 && data.address1.length > 2)
                            cs += data.address1 + '<br/>';
                        if (data.address2 && data.address2.length > 2)
                            cs += data.address2 + '<br/>';
                        if (data.zipcode && data.zipcode.length > 2)
                            cs += data.zipcode + ' ';
                        if (data.city && data.city.length > 2)
                            cs += data.city + '<br/>';
                        if (data.phone && data.phone.length > 2)
                            cs += 'Phone: ' + data.phone + '<br/>';
                        if (data.fax && data.fax.length > 2)
                            cs += 'Fax: ' + data.fax + '<br/>';
                        if (data.email && data.email.length > 2)
                            cs += 'E-Mail: <a href="mailto:' + data.email + '">' + data.email + '</a><br/>';
                        if (data.contactName && data.contactName.length > 2)
                            cs += '<br/>Primary Contact Name: <b>' + data.contactName + '</b><br/>';
                        if (data.contactEmail && data.contactEmail.length > 2)
                            cs += 'Primary Contact E-Mail: <a href="mailto:' + data.contactEmail + '">' + data.contactEmail + '</a><br/>';
                        if (data.additionalContacts && data.additionalContacts.length > 2)
                            cs += '<br/>' + data.additionalContacts + '<br/>';
                        cs += '</div>';

                        var centerMarker = L.marker([data.latitude, data.longitude]).bindPopup(cs);
                        centerMarkers.push(centerMarker);
                    }

                    //console.log("markers array: ", centerMarkers);
                    tcLayer = L.layerGroup(centerMarkers);
                    //console.log("layer: ", tcLayer);
                    map.addLayer(tcLayer);
                }
            });

        }

        function openExistingPopupAt(map, lat, lon) {
            // OK this is a bit of a hack because neither map.openPopup nor .openOn(map) works.
            // So we just create a new popup on the coordinates with the stored information.
            map.eachLayer(function (layer) {
                if (layer.hasOwnProperty("_popupHandlersAdded") && layer._popupHandlersAdded) {
                    if (lat === layer._latlng.lat && lon === layer._latlng.lng) {
                        //map.openPopup(layer._popup._content, [lat, lon]);
                        L.marker([lat, lon]).addTo(map).bindPopup(layer._popup._content).openPopup();
                        map.removeLayer(layer);
                    }
                }
            });
        }

        function panToLocationFromNav(tableDivId) {
            // Get the coordinates from the treatement center overview widget and fly to the location on the map
            var table = $(tableDivId);

            if (table) {
                table.click(function (e) {
                    var sLon = e.target.parentElement.getAttribute('data-lon');
                    var sLat = e.target.parentElement.getAttribute('data-lat');
                    //console.log("lon: " + lon + ", lat: " + lat);

                    if (sLat && sLon) {
                        var lon = parseFloat(sLon);
                        var lat = parseFloat(sLat);

                        //console.log("lon: " + lon + ", lat: " + lat);
                        map.flyTo(new L.LatLng(lat, lon), 13, {animation: true});
                        openExistingPopupAt(map, lat, lon);
                    }
                });
            }
        }

        //
        // Let's start the fun
        //
        initMap();

        if (startWithMap === "offices" && map !== null) {
