		function write_canvas(){
			var canvas = new canvasDrawer('canvasForTransparent', 'forcanvas');
			canvas.draw();
			jQuery('#forcanvas').hide();
		}

		function write_canvas2(){
			var canvas = new canvasDrawer('canvasForTransparent2', 'forcanvas2');
			canvas.draw();
			jQuery('#forcanvas').hide();

	function canvasDrawer(canvasElementID,textElementID){
		this.canvas_id = canvasElementID;
		this.canvas = document.getElementById(this.canvas_id);
		this.ctx = this.canvas.getContext("2d");
		this.text_to_d = jQuery('#'+textElementID).text();

		this.draw = function(){
			var oryginal_width = this.canvas.width;
			var scale_obj = this.pixelRatio();
			var scale_ratio = scale_obj.scale;
			var rev_scale = scale_obj.scale_rev;

			var vertical_margins = 20
			var canvas_width = this.canvas.width;
			var canvas_height = this.canvas.height;
			var canvas_height2 = jQuery('#'+this.canvas_id).css('height').replace('px','');

			this.ctx.fillStyle = "white";
			this.setFont(36);
			this.ctx.textBaseline = "middle";
			var text_width = this.calctextlength();
			var obj_scale = this.getWidthScale(text_width,oryginal_width - vertical_margins*2);
			this.ctx.scale(obj_scale.scale,obj_scale.scale);
			var text_width2 = this.calctextlength();
			this.drawWhiteRect(text_width + vertical_margins*2*obj_scale.revert_scale ,canvas_height2*obj_scale.revert_scale);

			this.ctx.save();
			this.ctx.globalCompositeOperation = 'destination-out';
			if(navigator.userAgent.toLowerCase().indexOf('firefox') > -1) {
				this.ctx.fillStyle = 'black';
			}
			this.drawText(20 *obj_scale.revert_scale, canvas_height / 2 *obj_scale.revert_scale * rev_scale)
			this.ctx.restore();
		}


		this.pixelRatio = function(){
			var scale_obj = {scale:1,scale_rev:1};
			if (window.devicePixelRatio) {

	            var CanvasWidth = jQuery('#'+this.canvas_id).attr('width');
	            var CanvasHeight = jQuery('#'+this.canvas_id).attr('height').replace('px','');
	            var CanvasCssWidth = CanvasWidth;
	            var CanvasCssHeight = CanvasHeight;

	            jQuery('#'+this.canvas_id).attr('width', CanvasWidth * window.devicePixelRatio);
	            jQuery('#'+this.canvas_id).attr('height', CanvasHeight * window.devicePixelRatio);
	            jQuery('#'+this.canvas_id).css('width', CanvasCssWidth);
	            jQuery('#'+this.canvas_id).css('height', CanvasCssHeight);
	            this.ctx.scale(window.devicePixelRatio, window.devicePixelRatio);

	            scale_obj.scale_rev = 1/window.devicePixelRatio;
	            scale_obj.scale = window.devicePixelRatio;
			}

			return scale_obj;
		}
		this.calctextlength = function(){
			var length = 0;
			var split = this.text_to_d.split('®');
			for( var i = 0; i < split.length-1; i++){
				length += this.ctx.measureText(split[i]).width;
				this.ctx.save();
					this.setFont(16);
					length += this.ctx.measureText('®').width;
				this.ctx.restore();
			}
			length += this.ctx.measureText(split[i]).width;
			return length;
		}
		this.drawText = function(left,bottom){

			var split = this.text_to_d.split('®');
			for( var i = 0; i < split.length-1; i++){
				this.ctx.fillText(split[i], left, bottom);
				left += this.ctx.measureText(split[i]).width;
				this.ctx.save();
					this.setFont(16);
					this.ctx.fillText('®', left+3, bottom-0.25*bottom);
					left += this.ctx.measureText('®').width;
				this.ctx.restore();
			}
			this.ctx.fillText(split[i], left, bottom);
		}

		this.drawWhiteRect =  function(width,height){
			var color = this.ctx.fillStyle;
			this.ctx.fillStyle = "white";
			this.ctx.fillRect(0,0,width,height);
			this.ctx.fillStyle = color;
		}
		this.setFont = function (size){
			this.ctx.font = size+"px 'Roboto', sans-serif";
			if(this.ctx.font != size+"px Roboto, sans-serif")
				this.ctx.font = size+"px Arial";
		}

		this.getWidthScale = function (text_width,width){
			var scale_obj = {scale:1,revert_scale:1};

			if(text_width > width){
				scale_obj.scale = width/text_width;
				scale_obj.revert_scale = text_width/width;
			}

			return scale_obj;
		}

	}
