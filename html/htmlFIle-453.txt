function rotateBanner(direction) {
    if (direction == 'prev') {
        if (jQuery('.home-banner.active').prev('.home-banner').length > 0) {
            jQuery('.home-banner.active').removeClass('active').css({zIndex: 0}).prev().css({zIndex: 1}).addClass('active');
        } else {
            jQuery('.home-banner.active').removeClass('active').css({zIndex: 0});
            jQuery('.home-banner:last-child').css({zIndex: 1}).addClass('active');
        } 
    } else {
        if (jQuery('.home-banner.active').next().length > 0) {
            jQuery('.home-banner.active').removeClass('active').css({zIndex: 0}).next().css({zIndex: 1}).addClass('active');
        } else {
            jQuery('.home-banner.active').removeClass('active').css({zIndex: 0});
            jQuery('.home-banner:nth-child(3)').css({zIndex: 1}).css({zIndex: 1}).addClass('active');
        }
    }
}

var bannerInterval = setInterval(rotateBanner, 8000);

jQuery('#home-banner-next').click(function(){
    rotateBanner();
    clearInterval(bannerInterval);
    bannerInterval = setInterval(rotateBanner, 8000);
});

jQuery('#home-banner-prev').click(function(){
    rotateBanner('prev');
    clearInterval(bannerInterval);
    bannerInterval = setInterval(rotateBanner, 8000);
});
