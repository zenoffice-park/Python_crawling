        dom = $(".cscSearchResult ul");
    clearTimeout(typingTimer);
    typingTimer = setTimeout(function(){
        value = value.replace("&","").replace("++","+");
        if(value.length > 2){
          $.ajax({

                  var ct = $(v).data('content-type')
                  if(ct !== undefined || ct !== ""){
                    $(v).addClass(iconClass(ct));
                  }
                } );
              }
            }
          })
        }
    }, doneTypingInterval);
  });
