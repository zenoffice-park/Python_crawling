/*プライバシーポリシーの有効・無効*/
function policy_chk_controll(){
	var enable_flg = inputChk(false);
	
	if(enable_flg){
		$('#policy_chk').prop('disabled',false);
	}else{
		$('#policy_chk').prop('disabled',true);
	}
	
	return true;
}

/**
 * 入力チェック
 * submit true:submit時（エラーメッセージを表示する） false:チェックの結果を返すだけ
 **/
function inputChk(submit){
	
	var return_flg = true;
	
	if(submit) $('#e_msg_box').hide();
	if(submit) $('.must_text_msg').remove();
	
	//テキストボックスの必須チェック（住所以外）
	$('.must_text').each(function(){
	  if($(this).val() == ''){
	    if(submit) $('#e_msg_box').show();
	    if(submit) $(this).parent().append('<span role="alert" class="must_text_msg wpcf7-not-valid-tip">必須項目に入力してください。</span>');
	    return_flg = false;
	  }
	});
	
	//テキストボックスの必須チェック（住所限定）
	var address_return_flg = true;
	$('.must_text_address').each(function(){
	  if($(this).val() == ''){
	    address_return_flg = false;
	  }
	});
	if(!address_return_flg){
	  if(submit) $('#e_msg_box').show();
	  if(submit) $('#last_address').parent().append('<span role="alert" class="must_text_msg wpcf7-not-valid-tip">必須項目に入力してください。</span>');
	  return_flg = false;
	}
	
	//チェックボックス
	$('.must_check').each(function(){
	  if($(this).find(':checkbox:checked').size() < 1){
	  	if(submit) $('#e_msg_box').show();
	    if(submit) $(this).append('<span role="alert" class="must_text_msg wpcf7-not-valid-tip">必須項目に入力してください。</span>');
	  	return_flg = false;
	  }
	});
	
	//メールアドレスの再入力チェック（submit時以外はチェックしない）
	if(submit){
	  if($('#your-email').val() != '' && $('#your-email2').val() != ''){
	  	if($('#your-email').val() != $('#your-email2').val()){
	  		$('#your-email, #your-email2').parent().append('<span role="alert" class="must_text_msg wpcf7-not-valid-tip">メールアドレスが一致しません。</span>');
	  		return_flg = false;
	  	}
	  }
	}
	
	//ポリシー（submit時以外はチェックしない）
	/*
	if(submit){
		if($('#policy_chk:checked').size() < 1){
			$('#policy_chk').parent().append('<span role="alert" class="must_text_msg wpcf7-not-valid-tip">必須項目に入力してください。</span>');
		  return_flg = false;
		}
	}
	*/
	
	//チェックボックスの値をhiddenにカンマ区切りで格納
	if(submit & return_flg){
	  chkToConmma('mailtype');     //お問い合わせの種類
	  chkToConmma('servicekaigo'); //介護保険
	  chkToConmma('servicesogo');  //総合支援
	  chkToConmma('question');     //今回、お問い合わせ・・・

	}
	
	return return_flg;
}

/**
 * チェックボックスの値をカンマ区切り
 * [A]チェックボックスのclass名、及びhiddenのid　※必ず統一した要素を作成すること
 **/
function chkToConmma(id_class_name){
	var str_value = '';
	
	$('.'+id_class_name + ':checked').each(function(){
	  str_value += (str_value == '') ? $(this).val(): ',' + $(this).val();
	});
	
	$('#h_'+id_class_name).val(str_value);
}
