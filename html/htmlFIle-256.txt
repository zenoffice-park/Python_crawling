$(function(){
	var smp_menu_list_nav = $('#smp_menu_list_nav'),
	switchPoint = 600,
	slideSpeed = 300,
	fadeSpeed = 500;

	var smp_menu_list_nav = smp_menu_list_nav.html();
	$(window).load(function(){
		/*リサイズ時に実行 PCサイズかスマホサイズか */
		function menu_resize_kirikae(){
			if(window.innerWidth <= switchPoint){
				if(!($('#smp_menu_div').length)){
					$('body').prepend('<div id="smp_menu_soto_div" onclick="menu_disp_showhide_open();" ></div><div id="switchBtnArea"><a href="javascript:void(0);" id="switchBtn"><span></span><span></span><span></span></a></div><div id="smp_menu_div"></div>');
					$('#smp_menu_div').append(smp_menu_list_nav);/**/
				}
			} else {
				$('#smp_menu_soto_div,#switchBtnArea,#smp_menu_div').remove();
				$('body').removeAttr('style');
			}
		}
		$(window).on('resize', function(){
			/*リサイズ時に実行*/
			menu_resize_kirikae();
		});
		menu_resize_kirikae();
	});
});
	/* スマホメニュー 表示非表示切り替え */
	function menu_disp_showhide_open(){
		var smp_menu_list_nav = $('#smp_menu_list_nav'),
		switchPoint = 600,
		slideSpeed = 300,
		fadeSpeed = 500;
		var btnLeft = parseInt($('#switchBtn').css('left'));
		var menuWidth =  $('#smp_menu_div').outerWidth();

		/* スマホメニュー 表示/非表示 */
		if($('#switchBtn').hasClass('btnClose')){
			$('#switchBtn').removeClass('btnClose').removeAttr('style');
			$('#smp_menu_soto_div').stop().animate({opacity:'0'},fadeSpeed,function(){
				$('#smp_menu_soto_div').removeAttr('style');
			});
			$('#smp_menu_div').stop().animate({left:'-' + menuWidth + 'px'},slideSpeed);
			/*$('body').removeAttr('style');*/
		} else {
			$('#switchBtn').addClass('btnClose').css({position:'fixed'}).stop().animate({left:menuWidth + btnLeft},slideSpeed);
			$('#smp_menu_soto_div').css({display:'block',opacity:'0'}).stop().animate({opacity:'1'},fadeSpeed);
			$('#smp_menu_div').stop().animate({left:'0'},slideSpeed);
			/*$('body').css({position:'fixed'});*/
		}
	}

		function cal_move(calId,m){
			if(m == null)
				calObj[calId].currentMonth = new Date(calObj.year+"/"+(calObj.month)+"/"+"1 00:00:00");
			else {
				var day = calObj[calId].currentMonth.getDate();
				var month = calObj[calId].currentMonth.getMonth() + 1;
				var year = calObj[calId].currentMonth.getFullYear();
				if(0 < month + m && month + m < 13)
					calObj[calId].currentMonth = new Date(year+"/"+(month + m)+"/"+"1 00:00:00");
				else if((month + m) < 1){
					year--;
					month = 12;
					calObj[calId].currentMonth = new Date(year+"/"+(month)+"/"+"1 00:00:00");
				}
				else {
					year++;
					month = 1;

				}
			}
			cal_create(calId);
		}
		function cal_clone(obj) {
			var dest;
			if(typeof obj == 'object'){
				if(obj instanceof Array){
					dest = new Array();
					for(i=0;i<obj.length;i++)
						dest[i] = cal_clone(obj[i]);
				}
				else {
					dest = new Object();
					for(prop in obj)
						dest[prop] = cal_clone(obj[prop]);
				}
			}
			else
				dest = obj;
			return dest;
		}
