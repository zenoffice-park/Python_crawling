$(document).ready(function() {
    if ( $('#docdl').length > 0 ) {
        $("input[type=checkbox]").change(function() {
            if ( $(this).parent().prop("tagName").toLowerCase() === "th" ) {
                if ( $(this).is(':checked') ) { //Check all checkbox
                    $(this).parents("table").find("td input").each(function() {
                        $(this).prop("checked", true);
                    });
                } else {
                    $(this).parents("table").find("td input").each(function() {
                        $(this).prop("checked", false);
                    });
                }
            }
        });

        $('#docdl').unbind('click').click(function(e) {
            e.preventDefault();
            var checked_element = new Array();
            $("input[type=checkbox]:checked").each(function() {
                //checked_element.push($(this).attr("id"));
                if ( $(this).val() !== "" ) {
                    checked_element.push($(this).val());
                }
            });

            fileid_param_remote = checked_element.join("+");
            fileid_param_local = checked_element.join(":");

            filedlurl = "https://www.sbcloud.co.jp/docdl/?docids=" + fileid_param_local;

            url = $(this).attr("href").replace("IDS", fileid_param_remote).replace("DLURL", filedlurl);

            window.open(url);

            return false;
        });
    }
});
