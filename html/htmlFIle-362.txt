<script type="text/javascript">
// sets the cookie and if we close all browsers and the page will pop up a new window
function setCookieForPopup(cname,cvalue) { 
    var d = new Date();
    d.setTime(d.getTime() + (24*60*60*1000));
    var expires = "expires="+d.toUTCString();     
    document.cookie = cname+"="+cvalue+";"+expires+ "; path=/jp";
    document.cookie = cname+"="+cvalue+";"+expires+ "; path=/wps/wcm/myconnect/jp";
}

// returns the cookie, if it exists...
function getCookieForPopup(cname) {
    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i=0; i<ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1);
        if (c.indexOf(name) == 0) {
            return c.substring(name.length, c.length);
        }
    }
    return "";
}

// checks if its there, if not will drop the cookie

    // so the markup will have the popup, but will have the hide-me class
    var path = window.location.pathname;
    var path= path.toLowerCase();

            buildPopupForPopup();
        }
    }
}
