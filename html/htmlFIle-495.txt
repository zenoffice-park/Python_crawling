<script type="text/javascript">
    $(document).ready(function () {
        var languageMenuContainer = document.getElementById('language-menu-container');
        var languageSelector = document.getElementById('language-selector');

        languageMenuContainer.style.bottom = languageSelector.offsetHeight - 15 + "px";
        languageMenuContainer.style.right = "0px";

        // Clone the mobile menu, make sure it's hidden, and attach it after the footer
        var mobileMenu = $('#language-mobile-menu').clone();
        mobileMenu.css('display', 'none');
        mobileMenu.insertAfter('.footer');

        // Remove the now copied mobile menu code
        $('#language-mobile-menu').remove();

        $(".language-hovermenu-toggle").each(function () {
            var hovermenu_class = $(this).parent().attr("data-toggle");

            var hovering_main_menu_item = false;
            var hovering_hover_menu = false;

            var hovermenu_update = function () {
                if (!hovering_main_menu_item && !hovering_hover_menu) {
                    $("." + hovermenu_class).hide();
                } else {
                    $("." + hovermenu_class).show();
                }
            };

            $("." + hovermenu_class).hide();

            $(this).click(function () {
                // Activate the horizontal language selector only when in desktop mode
                if ($(window).width() >= 768) {
                    $('#language-mobile-menu').css('display', 'none');
                    
                    hovering_main_menu_item = true;
                    hovermenu_update();

                    $("." + hovermenu_class).mouseover(function () {
                        hovering_hover_menu = true;
                        hovermenu_update();
                    });

                    $("." + hovermenu_class).mouseout(function () {
                        hovering_hover_menu = false;
                        hovermenu_update();
                    });
                } else if ($('#language-mobile-menu').css('display') == 'none') {
                    var mobileMenu = $('#language-mobile-menu');
                    mobileMenu.css('display', 'block');
                    $("html, body").animate({ scrollTop: $(document).height() }, "slow");
                } else {
                    var mobileMenu = $('#language-mobile-menu');
                    mobileMenu.css('display', 'none');
                }
            });

            $(document).mouseup(function (e) {
                var container = $(".language-hovermenu-toggle");

                if (!container.is(e.target) // if the target of the click isn't the container...
                    && container.has(e.target).length === 0) // ... nor a descendant of the container
                {
                    hovering_main_menu_item = false;
                    hovermenu_update();
                }
            });
        });
    });
</script>
