function validateNumric(event) {
	// Allow: backspace, delete, tab, escape, enter and .
	if ( $.inArray(event.keyCode,[46,8,9,27,13,190]) !== -1 ||
		// Allow: Ctrl+A
		(event.keyCode == 65 && event.ctrlKey === true) || 
		// Allow: home, end, left, right
		(event.keyCode >= 35 && event.keyCode <= 39)) {
		// let it happen, don't do anything
		return;
	} else {
		// Ensure that it is a number and stop the keypress
		if (event.shiftKey || (event.keyCode < 48 || event.keyCode > 57) && (event.keyCode < 96 || event.keyCode > 105 )) {
			event.preventDefault(); 
		}
	}
}

        return String.fromCharCode(s.charCodeAt(0) - 0xFEE0);
    });
    return str.replace(/[^0-9]/g,"");
}

function updateSumPlice() {
	var sumvalue_init = 0;
	var sumvalue_month = 0;
	
	if (isTypeOnpre()) {
		sumvalue_init = sumOnpreInit();
		sumvalue_month = 0;
	} else {
		sumvalue_init = sumCloudInit();
		sumvalue_month = sumCloudMonth();
	}
	
	var init_str = String( sumvalue_init ).replace( /(\d)(?=(\d\d\d)+(?!\d))/g, '$1,' );
	var month_str = String( sumvalue_month ).replace( /(\d)(?=(\d\d\d)+(?!\d))/g, '$1,' );
	$("#"+ID_SUMPLICE_INIT).attr('value', init_str);
	$("#"+ID_SUMPLICE_MONTH).attr('value', month_str);
}

function sumOnpreInit() {
	// ライセンス料金
	var pbx_sheet_am = 1;
	var pbx_sheet_value = 29800;
	var callcenter_sheet_am = 1;
	var callcenter_sheet_value = 69800;
	// サーバー料金
	// var server_l_value = 850000;
	// var server_m_value = 750000;
	// var server_s_value = 500000;
	var server_s_value = 0;
	// 構築費
	// var cons_value = 900000;
	var cons_value = 0;
	// 機能料金-IVR
	var ivr_am = 1;
	var ivrch_value = 45000;
	// 機能料金-MIS
	var mis_am = 2;
	var mis_value = 396000;
	// 機能料金-AUTOCALL
	var autocall_am = 1;
	var autocall_value = 798000;
	// 同時通話料金
	var maxcall_am = 1;
	var maxcall_value = 1111;
	// 電話番号料金
	var maxtelnum_am = 1;
	var maxtelnum_value = 2222;
	
	var sumvalue = 0;
	// ライセンス料金
	if (isEditionPbx()) {
		sumvalue += getUnitPrice(getSheet(), pbx_sheet_am, pbx_sheet_value);
	} else {
		sumvalue += getUnitPrice(getSheet(), callcenter_sheet_am, callcenter_sheet_value);
	}
	// サーバー料金
	sumvalue += server_s_value;
	// 構築費
	sumvalue += cons_value;
	// 機能料金-IVR
	if (isIvrUse()) {
		sumvalue += getUnitPrice(getIvrCh(), ivr_am, ivrch_value);
	}
	// 機能料金-MIS
	if (isMisUse()) {
		sumvalue += getUnitPrice(mis_am, mis_am, mis_value);
	}
	// 機能料金-AUTOCALL
	if (isAutocallUse()) {
		sumvalue += getUnitPrice(autocall_am, autocall_am, autocall_value);
	}
	
	return sumvalue;
}

function sumCloudInit() {
	//　契約初期料金
	var cloud_l_am = 48;
	var cloud_l_value = 80000;
	var cloud_m_am = 13;
	var cloud_m_value = 60000;
	var cloud_s_am = 1;
	var cloud_s_value = 40000;
	// 回線初期設定費
	var line_init = 1500;
	//var line_init = 0;
	// 電話番号初期設定費
	var number_init = 150;
	// ch初期設定費（同時通話初期設定費）
	var ch_init = 2000;
	// 選択番号通知工事費
	var number_cons = 0;
	// カスタマコントロール初期設定費
	var cuscon_init = 0;

	var sumvalue = 0;
	// 契約初期料金
	if (cloud_l_am < getSheet()) {
		sumvalue += cloud_l_value;
	} else if (cloud_m_am <= getSheet()) {
		sumvalue += cloud_m_value;
	} else if (cloud_s_am <= getSheet()) {
		sumvalue += cloud_s_value;
	} else {
		sumvalue += cloud_s_value;
	}

	// 回線初期設定費
	sumvalue += line_init;
	// 電話番号初期設定費
	sumvalue += (number_init * getMaxtelnum());
	// ch初期設定費（同時通話初期設定費）
	sumvalue += (ch_init * getMaxcall());
	// 選択番号通知工事費
	sumvalue += (number_cons * getMaxtelnum());
	// カスタマコントロール初期設定費
	sumvalue += cuscon_init;

	return sumvalue;
}
